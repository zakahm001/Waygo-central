console.log('Connected to Waygo live'); }); // Driver positions on map const mapDrivers = {}; socket.on('driver:location', (data) => { const { driverId, lat, lng } = data; updateDriverOnMap(driverId, lat, lng); }); // New booking arrives socket.on('booking:new', (booking) => { showToast('🚖 Ny bokning: ' + booking.booking_ref); loadDashboard(); }); // Booking assigned socket.on('booking:assigned', () => { loadDashboard(); }); // Booking cancelled socket.on('booking:cancelled', () => { loadDashboard(); }); // Cancel request from customer socket.on('booking:cancelRequest', (data) => { showToast('⚠️ Avbokningsförfrågan: ' + data.customerName); loadDashboard(); }); // ── MAP FUNCTIONS ── function updateDriverOnMap( driverId, lat, lng) { const map = document.getElementById( 'liveMap'); if (!map) return; // Convert GPS to map percentage // Stockholm area bounds const minLat = 59.2; const maxLat = 59.5; const minLng = 17.8; const maxLng = 18.2; const x = ((lng - minLng) / (maxLng - minLng)) * 90 + 5; const y = ((maxLat - lat) / (maxLat - minLat)) * 90 + 5; let dot = document.getElementById( 'mapDriver' + driverId); if (!dot) { dot = document.createElement('div'); dot.id = 'mapDriver' + driverId; dot.className = 'map-driver-dot'; dot.innerHTML = '🚖'; dot.title = 'Förare ' + driverId; map.appendChild(dot); } dot.style.left = x + '%'; dot.style.top = y + '%'; // Update active count const count = map.querySelectorAll( '.map-driver-dot').length; const countEl = document.getElementById( 'activeCount'); if (countEl) countEl.textContent = count; } function addStaticDriversToMap(drivers) { drivers.forEach((d, i) => { if (d.status === 'offline') return; const positions = [ { x: 45, y: 35 }, { x: 25, y: 60 }, { x: 70, y: 65 }, ]; const pos = positions[i] || { x: 50, y: 50 }; const map = document.getElementById( 'liveMap'); if (!map) return; let dot = document.getElementById( 'mapDriver' + d.id); if (!dot) { dot = document.createElement('div'); dot.id = 'mapDriver' + d.id; dot.className = 'map-driver-dot ' + d.status; const label = document.createElement( 'span'); label.className = 'map-label-tag'; label.textContent = d.name.split(' ')[0]; dot.innerHTML = d.status === 'busy' ? '🚖' : '🚕'; dot.appendChild(label); map.appendChild(dot); } dot.style.left = pos.x + '%'; dot.style.top = pos.y + '%'; }); const active = drivers.filter( d => d.status !== 'offline').length; const countEl = document.getElementById( 'activeCount'); if (countEl) countEl.textContent = active; } // ── CHECK LOGIN ── if (localStorage.getItem( 'waygo_logged_in') !== 'true') { window.location.href = 'index.html'; } // ── LOGOUT ── function logout() { localStorage.removeItem( 'waygo_logged_in'); window.location.href = 'index.html'; } // ── SHOW TOAST ── function showToast(msg) { const t = document.getElementById( 'toast'); t.textContent = msg; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 3000); } // ── GET BADGE HTML ── function getBadge(status) { const badges = { assigned: `Tilldelad`, pending: `Väntar`, cancelled: `Avbokad`, }; return badges[status] || `${status} `; } // ── ASSIGN DRIVER ── async function assignDriver( bookingId, driverId, driverName) { try { const res = await fetch( `${API}/bookings/${bookingId}/assign`, { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ driver_id: driverId }) } ); if (res.ok) { showToast( `✓ Tilldelad till ${driverName}` ); loadDashboard(); } } catch (err) { showToast('⚠️ Fel vid tilldelning'); } } // ── CANCEL BOOKING ── async function cancelBooking(id, ref) { if (!confirm( `Avboka ${ref}?`)) return; try { const res = await fetch( `${API}/bookings/${id}/cancel`, { method: 'PATCH' } ); if (res.ok) { showToast(`✓ ${ref} avbokad`); loadDashboard(); } } catch (err) { showToast('⚠️ Fel vid avbokning'); } } // ── BUILD BOOKING CARD ── function buildCard(b, drivers) { const isUrgent = b.cancel_request; const availableDrivers = drivers .filter(d => d.status === 'available'); let assignButtons = ''; if (b.status === 'pending' && availableDrivers.length > 0) { assignButtons = availableDrivers .map(d => ` `).join(''); } return `
${b.booking_ref} ${getBadge(b.status)}
${b.customer_name}
📍 ${b.from_address} → 🏁 ${b.to_address}
${b.driver_name ? '🚖 ' + b.driver_name : 'Ingen förare ännu'} ${b.fare_sek} kr
${assignButtons} ${b.status !== 'cancelled' ? `` : ''}
`; } // ── LOAD DASHBOARD ── async function loadDashboard() { try { // Fetch bookings and drivers const [bRes, dRes] = await Promise.all([ fetch(`${API}/bookings`), fetch(`${API}/drivers`) ]); const bData = await bRes.json(); const dData = await dRes.json(); const bookings = bData.bookings; const drivers = dData.drivers; addStaticDrivers ToMap(drivers); // Update stats document.getElementById( 'statActive').textContent = bookings.filter(b => b.status === 'assigned' ).length; document.getElementById( 'statPending').textContent = bookings.filter(b => b.status === 'pending' ).length; document.getElementById( 'statDrivers').textContent = drivers.filter(d => d.status === 'available' ).length; document.getElementById( 'statCancel').textContent = bookings.filter(b => b.cancel_request ).length; // Build booking cards const active = bookings.filter( b => b.status === 'assigned'); const pending = bookings.filter( b => b.status === 'pending'); const cancelled = bookings.filter( b => b.status === 'cancelled'); let html = ''; if (active.length > 0) { html += `
Aktiva resor
`; html += active.map(b => buildCard(b, drivers)).join(''); } if (pending.length > 0) { html += `
Väntar på förare
`; html += pending.map(b => buildCard(b, drivers)).join(''); } if (cancelled.length > 0) { html += `
Avbokade
`; html += cancelled.map(b => buildCard(b, drivers)).join(''); } if (bookings.length === 0) { html = `
Inga bokningar ännu
`; } document.getElementById( 'bookingsList').innerHTML = html; // Hide error box document.getElementById( 'errorBox').style.display = 'none'; } catch (err) { // Show error document.getElementById( 'errorBox').style.display = 'block'; document.getElementById( 'bookingsList').innerHTML = `
Kunde inte ladda bokningar.
`; } } // ── START ── loadDashboard(); // ── AUTO REFRESH EVERY 30 SECONDS ── setInterval(loadDashboard, 30000);
